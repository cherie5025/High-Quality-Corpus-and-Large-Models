一、数据库表结构
1. papers 表（文献主表）
sql
CREATE TABLE papers (
    id INT PRIMARY KEY AUTO_INCREMENT,
    title TEXT,                           -- 文献标题
    authors TEXT,                         -- 作者（原始格式）
    corresponding_author VARCHAR(200),    -- 通讯作者
    journal VARCHAR(200),                 -- 期刊
    publication_date VARCHAR(20),         -- 发表日期
    document_type VARCHAR(20),            -- 文献类型：综述/研究型/会议论文
    impact_factor FLOAT,                  -- 影响因子
    citation_count INT,                   -- 被引次数
    abstract TEXT,                        -- 摘要
    keywords TEXT,                        -- 关键词
    chemical_system VARCHAR(100),         -- 化学体系
    _source VARCHAR(10) DEFAULT 'csv',    -- 数据来源：csv/manual/pdf
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    -- 可信度计算结果（由 Python 脚本更新）
    C_time FLOAT,                         -- 时效可信度
    C_print FLOAT,                        -- 期刊可信度
    C_author FLOAT,                       -- 作者可信度
    C_overall FLOAT,                      -- 综合可信度
    w_time FLOAT,                         -- 时效权重
    w_print FLOAT,                        -- 期刊权重
    w_author FLOAT,                       -- 作者权重
    wa1 FLOAT, wa2 FLOAT, wa3 FLOAT, wa4 FLOAT  -- 作者网络指标权重



1. author_acc 表（作者影响力值）
sql
CREATE TABLE author_acc (
    author_name VARCHAR(200) PRIMARY KEY,  -- 标准化作者名
    acc FLOAT                              -- 作者综合影响力
);


二、API 接口列表（共 10 个）
基础 URL: http://localhost:3007
方法	路径	说明	请求参数
POST	/api/upload	上传 Excel 文件	file (multipart/form-data)
POST	/api/manual	手动录入文献	JSON: {title, authors, corresponding, abstract, keywords}
POST	/api/upload-pdf	上传 PDF	file (multipart/form-data)
GET	/api/data	CSV 文献列表（分页）	?page=1&pageSize=10
GET	/api/manual-data	手动录入列表（分页）	?page=1&pageSize=10
GET	/api/id/:id	单篇文献详情	路径参数: id
GET	/api/author-contribution/:id	作者贡献度表格	路径参数: id
GET	/api/date-range	文献日期范围	无
GET	/api/network-data	作者合作网络数据	无
POST	/api/compute-all	触发批量计算	无
